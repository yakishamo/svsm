56#[derive(Debug)]
57pub struct KernelHeap {
58    virt_base: VirtAddr,
59    phys_base: PhysAddr,
60    page_count: usize,
61    next_free: usize,
62}
63
64impl KernelHeap {
65    pub fn create(virt_base: VirtAddr, prange: MemoryRegion<PhysAddr>) -> Self {
66        Self {
67            virt_base,
68            phys_base: prange.start(),
69            page_count: prange.len() / PAGE_SIZE,
70            next_free: 0,
71        }
72    }
73
74    pub fn virt_base(&self) -> VirtAddr {
75        self.virt_base
76    }
77
78    pub fn phys_base(&self) -> PhysAddr {
79        self.phys_base
80    }
81
82    pub fn page_count(&self) -> usize {
83        self.page_count
84    }
85
86    pub fn next_free(&self) -> usize {
87        self.next_free
88    }
89
90    pub fn allocate(&mut self, size: usize) -> Result<(VirtAddr, PhysAddr), SvsmError> {
91        let page_count = round_to_pages(size);
92        self.allocate_pages(page_count)
93    }
94
95    pub fn allocate_pages(&mut self, page_count: usize) -> Result<(VirtAddr, PhysAddr), SvsmError> {
96        // Verify that this allocation will not overflow the heap.
97        let next_free = self.next_free + page_count;
98        if next_free > self.page_count {
99            return Err(AllocError::OutOfMemory.into());
100        }
101
102        // Calculate the allocation base based on the current position within
103        // the heap.
104        let offset = self.next_free * PAGE_SIZE;
105        let virt_addr = self.virt_base + offset;
106        let phys_addr = self.phys_base + offset;
107
108        // Move the allocation cursor beyond this allocation.
109        self.next_free = next_free;
110        Ok((virt_addr, phys_addr))
111    }
112}

114fn setup_stage2_allocator(heap_start: u64, heap_end: u64) {
115    let vstart = VirtAddr::from(heap_start);
116    let vend = VirtAddr::from(heap_end);
117    let pstart = PhysAddr::from(vstart.bits()); // Identity mapping
118    let nr_pages = (vend - vstart) / PAGE_SIZE;
119
120    root_mem_init(pstart, vstart, nr_pages, 0);
121}

123fn init_percpu(platform: &mut dyn SvsmPlatform) -> Result<(), SvsmError> {
124    // SAFETY: this is the first CPU, so there can be no other dependencies
125    // on multi-threaded access to the per-cpu areas.
126    let percpu_shared = unsafe { PERCPU_AREAS.create_new(0) };
127    let bsp_percpu = PerCpu::alloc(percpu_shared)?;
128    bsp_percpu.set_current_stack(MemoryRegion::from_addresses(
129        VirtAddr::from(STAGE2_STACK_END as u64),
130        VirtAddr::from(STAGE2_STACK as u64),
131    ));
132    // SAFETY: pgtable is properly aligned and is never freed within the
133    // lifetime of stage2. We go through a raw pointer to promote it to a
134    // static mut. Only the BSP is able to get a reference to it so no
135    // aliasing can occur.
136    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
137    bsp_percpu.set_pgtable(init_pgtable);
138    bsp_percpu.map_self_stage2()?;
139    platform.setup_guest_host_comm(bsp_percpu, true);
140    Ok(())
141}

143/// Release all resources in the `PerCpu` instance associated with the current
144/// CPU.
145///
146/// # Safety
147///
148/// The caller must ensure that the `PerCpu` is never used again.
149unsafe fn shutdown_percpu() {
150    let ptr = SVSM_PERCPU_BASE.as_mut_ptr::<PerCpu>();
151    // SAFETY: ptr is properly aligned but the caller must ensure the PerCpu
152    // structure is valid and not aliased.
153    unsafe {
154        core::ptr::drop_in_place(ptr);
155    }
156    // SAFETY: pgtable is properly aligned and is never freed within the
157    // lifetime of stage2. We go through a raw pointer to promote it to a
158    // static mut. Only the BSP is able to get a reference to it so no
159    // aliasing can occur.
160    let init_pgtable = unsafe { (&raw mut pgtable).as_mut().unwrap() };
161    init_pgtable.unmap_4k(SVSM_PERCPU_BASE);
162    flush_tlb_percpu();
163}

165// SAFETY: the caller must guarantee that the IDT specified here will remain
166// in scope until a new IDT is loaded.
167unsafe fn setup_env(
168    config: &SvsmConfig<'_>,
169    platform: &mut dyn SvsmPlatform,
170    launch_info: &Stage2LaunchInfo,
171    cpuid_vaddr: Option<VirtAddr>,
172    idt: &mut IDT<'_>,
173) {
174    GLOBAL_GDT.load_selectors();
175    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
176    unsafe {
177        early_idt_init_no_ghcb(idt);
178    }
179
180    let debug_serial_port = config.debug_serial_port();
181    install_console_logger("Stage2").expect("Console logger already initialized");
182    platform
183        .env_setup(debug_serial_port, launch_info.vtom.try_into().unwrap())
184        .expect("Early environment setup failed");
185
186    let kernel_mapping = FixedAddressMappingRange::new(
187        VirtAddr::from(u64::from(STAGE2_START)),
188        VirtAddr::from(u64::from(launch_info.stage2_end)),
189        PhysAddr::from(u64::from(STAGE2_START)),
190    );
191
192    if let Some(cpuid_addr) = cpuid_vaddr {
193        // SAFETY: the CPUID page address specified in the launch info was
194        // mapped by the loader, which promises to supply a correctly formed
195        // CPUID page at that address.
196        let cpuid_page = unsafe { &*cpuid_addr.as_ptr::<SnpCpuidTable>() };
197        register_cpuid_table(cpuid_page);
198    }
199
200    paging_init(platform, true).expect("Failed to initialize early paging");
201
202    // Use the low 640 KB of memory as the heap.
203    let lowmem_region =
204        MemoryRegion::from_addresses(VirtAddr::from(0u64), VirtAddr::from(u64::from(LOWMEM_END)));
205    let heap_mapping = FixedAddressMappingRange::new(
206        lowmem_region.start(),
207        lowmem_region.end(),
208        PhysAddr::from(0u64),
209    );
210    init_kernel_mapping_info(kernel_mapping, Some(heap_mapping));
211
212    // Now that the heap virtual-to-physical mapping has been established,
213    // validate the first 640 KB of memory so it can be used if necessary.
214    // SAFETY: the low memory region is known not to overlap any memory in use.
215    unsafe {
216        platform
217            .validate_virtual_page_range(lowmem_region, PageValidateOp::Validate)
218            .expect("failed to validate low 640 KB");
219    }
220
221    // SAFETY: ap_flag is an extern static and this is the only place where we
222    // get a reference to it.
223    unsafe {
224        // Allow APs to proceed as the environment is now ready.
225        //
226        // Although APs use non-atomic loads in the ap_wait_for_env spin loop,
227        // the language and architectural guarantees of this atomic store (e.g.
228        // the compiler cannot move the previous stores past this atomic
229        // store-release, and x86 is a strongly-ordered system) make setting
230        // this flag more deterministic.
231        ap_flag.store(1, Ordering::Release);
232    }
233
234    // Configure the heap.
235    setup_stage2_allocator(STAGE2_HEAP_START.into(), STAGE2_HEAP_END.into());
236
237    init_percpu(platform).expect("Failed to initialize per-cpu area");
238
239    // Init IDT again with handlers requiring GHCB (eg. #VC handler)
240    // Must be done after init_percpu() to catch early #PFs
241    //
242    // SAFETY: the caller guarantees that the lifetime of this IDT is suitable.
243    unsafe {
244        early_idt_init(idt);
245    }
246
247    // Complete initializtion of the platform.  After that point, the console
248    // will be fully working and any unsupported configuration can be properly
249    // reported.
250    platform
251        .env_setup_late(debug_serial_port)
252        .expect("Late environment setup failed");
253
254    if cpuid_vaddr.is_some() {
255        dump_cpuid_table();
256    }
257}

259/// Map and validate the specified virtual memory region at the given physical
260/// address.
261/// # Safety
262/// The caller is required to ensure the safety of the virtual address range.
263unsafe fn map_and_validate(
264    platform: &dyn SvsmPlatform,
265    config: &SvsmConfig<'_>,
266    vregion: MemoryRegion<VirtAddr>,
267    paddr: PhysAddr,
268) -> Result<(), SvsmError> {
269    let flags = PTEntryFlags::PRESENT
270        | PTEntryFlags::WRITABLE
271        | PTEntryFlags::ACCESSED
272        | PTEntryFlags::DIRTY;
273
274    let mut pgtbl = this_cpu().get_pgtable();
275    pgtbl.map_region(vregion, paddr, flags)?;
276
277    if config.page_state_change_required() {
278        platform.page_state_change(
279            MemoryRegion::new(paddr, vregion.len()),
280            PageSize::Huge,
281            PageStateChangeOp::Private,
282        )?;
283    }
284    // SAFETY: the caller has ensured the safety of the virtual address range.
285    unsafe {
286        platform.validate_virtual_page_range(vregion, PageValidateOp::Validate)?;
287    }
288    valid_bitmap_set_valid_range(paddr, paddr + vregion.len());
289    Ok(())
290}

292#[inline]
293fn check_launch_info(launch_info: &KernelLaunchInfo) {
294    let offset: u64 = launch_info.heap_area_virt_start - launch_info.heap_area_phys_start;
295    let align: u64 = PAGE_SIZE_2M.try_into().unwrap();
296
297    assert!(is_aligned(offset, align));
298}

300/// Loads a single ELF segment and returns its virtual memory region.
301/// # Safety
302/// The caller is required to supply an appropriate virtual address for this
303/// ELF segment.
304unsafe fn load_elf_segment(
305    segment: elf::Elf64ImageLoadSegment<'_>,
306    paddr: PhysAddr,
307    platform: &dyn SvsmPlatform,
308    config: &SvsmConfig<'_>,
309) -> Result<MemoryRegion<VirtAddr>, SvsmError> {
310    // Find the segment's bounds
311    let segment_start = VirtAddr::from(segment.vaddr_range.vaddr_begin);
312    let segment_end = VirtAddr::from(segment.vaddr_range.vaddr_end).page_align_up();
313    let segment_len = segment_end - segment_start;
314    let segment_region = MemoryRegion::new(segment_start, segment_len);
315
316    // All ELF segments should be aligned to the page size. If not, there's
317    // the risk of pvalidating a page twice, bail out if so. Note that the
318    // ELF reading code had already verified that the individual segments,
319    // with bounds specified as in the ELF file, are non-overlapping.
320    if !segment_start.is_page_aligned() {
321        return Err(SvsmError::Elf(ElfError::UnalignedSegmentAddress));
322    }
323
324    // Map and validate the segment at the next contiguous physical address
325    // SAFETY: the caller has verified the safety of this virtual address
326    // region.
327    unsafe {
328        map_and_validate(platform, config, segment_region, paddr)?;
329    }
330
331    // Copy the segment contents and pad with zeroes
332    // SAFETY: the caller guarantees the correctness of the ELF segment's
333    // virtual address range.
334    let segment_buf =
335        unsafe { slice::from_raw_parts_mut(segment_start.as_mut_ptr::<u8>(), segment_len) };
336    let contents_len = segment.file_contents.len();
337    segment_buf[..contents_len].copy_from_slice(segment.file_contents);
338    segment_buf[contents_len..].fill(0);
339
340    Ok(segment_region)
341}

343/// Loads the kernel ELF and returns the virtual memory region where it
344/// resides, as well as its entry point. Updates the used physical memory
345/// region accordingly.
346fn load_kernel_elf(
347    launch_info: &Stage2LaunchInfo,
348    loaded_phys: &mut MemoryRegion<PhysAddr>,
349    platform: &dyn SvsmPlatform,
350    config: &SvsmConfig<'_>,
351) -> Result<(VirtAddr, MemoryRegion<VirtAddr>), SvsmError> {
352    // Find the bounds of the kernel ELF and load it into the ELF parser
353    let elf_start = PhysAddr::from(launch_info.kernel_elf_start as u64);
354    let elf_end = PhysAddr::from(launch_info.kernel_elf_end as u64);
355    let elf_len = elf_end - elf_start;
356    // SAFETY: the base address of the ELF image was selected by the loader and
357    // is known not to conflict with any other virtual address mappings.
358    let bytes = unsafe { slice::from_raw_parts(elf_start.bits() as *const u8, elf_len) };
359    let elf = elf::Elf64File::read(bytes)?;
360
361    let vaddr_alloc_info = elf.image_load_vaddr_alloc_info();
362    let vaddr_alloc_base = vaddr_alloc_info.range.vaddr_begin;
363
364    // Map, validate and populate the SVSM kernel ELF's PT_LOAD segments. The
365    // segments' virtual address range might not necessarily be contiguous,
366    // track their total extent along the way. Physical memory is successively
367    // being taken from the physical memory region, the remaining space will be
368    // available as heap space for the SVSM kernel. Remember the end of all
369    // physical memory occupied by the loaded ELF image.
370    let mut load_virt_start = None;
371    let mut load_virt_end = VirtAddr::null();
372    for segment in elf.image_load_segment_iter(vaddr_alloc_base) {
373        // SAFETY: the virtual address is calculated based on the base address
374        // of the image and the previously loaded segments, so it is correct
375        // for use.
376        let region = unsafe { load_elf_segment(segment, loaded_phys.end(), platform, config)? };
377        // Remember the mapping range's lower and upper bounds to pass it on
378        // the kernel later. Note that the segments are being iterated over
379        // here in increasing load order.
380        if load_virt_start.is_none() {
381            load_virt_start = Some(region.start());
382        }
383        load_virt_end = region.end();
384
385        // Update to the next contiguous physical address
386        *loaded_phys = loaded_phys.expand(region.len());
387    }
388
389    let Some(load_virt_start) = load_virt_start else {
390        log::error!("No loadable segment found in kernel ELF");
391        return Err(SvsmError::Mem);
392    };
393
394    // Apply relocations, if any
395    if let Some(dyn_relocs) =
396        elf.apply_dyn_relas(elf::Elf64X86RelocProcessor::new(), vaddr_alloc_base)?
397    {
398        for reloc in dyn_relocs {
399            let Some(reloc) = reloc? else {
400                continue;
401            };
402            // SAFETY: the relocation address is known to be correct. The ELF loader rejects
403            // relocations that point outside a PT_LOAD segment.
404            let dst = unsafe { slice::from_raw_parts_mut(reloc.dst as *mut u8, reloc.value_len) };
405            let src = &reloc.value[..reloc.value_len];
406            dst.copy_from_slice(src)
407        }
408    }
409
410    let entry = VirtAddr::from(elf.get_entry(vaddr_alloc_base));
411    let region = MemoryRegion::from_addresses(load_virt_start, load_virt_end);
412    Ok((entry, region))
413}

415/// Loads the IGVM params.  Returns the virtual and physical memory regions
416/// containing the loaded data.
417/// # Safety
418/// Ther caller is required to specify the correct virtual address for the
419/// kernel virtual region.
420fn load_igvm_params(
421    kernel_heap: &mut KernelHeap,
422    config: &SvsmConfig<'_>,
423    launch_info: &Stage2LaunchInfo,
424) -> Result<VirtAddr, SvsmError> {
425    let params = config.get_igvm_params();
426    let params_size = params.size();
427
428    // Allocate space in the kernel area to hold the parameters.
429    let (vaddr, _) = kernel_heap.allocate(params_size)?;
430
431    // Copy the contents over
432    let src_addr = VirtAddr::from(launch_info.igvm_params as u64);
433    // SAFETY: the source address specified in the launch info was mapped
434    // by the loader, which promises to supply a correctly formed IGRM
435    // parameter block
436    let igvm_src = unsafe { slice::from_raw_parts(src_addr.as_ptr::<u8>(), params_size) };
437    // SAFETY: the destination address came from the heap allocation above and
438    // can be used safely.
439    let igvm_dst = unsafe { slice::from_raw_parts_mut(vaddr.as_mut_ptr::<u8>(), params_size) };
440    igvm_dst.copy_from_slice(igvm_src);
441
442    Ok(vaddr)
443}

445/// Maps any remaining memory between the end of the kernel image and the end
446/// of the allocated kernel memory region as heap space. Exclude any memory
447/// reserved by the configuration.
448///
449/// # Panics
450///
451/// Panics if the allocated kernel region (`kernel_region`) is not sufficient
452/// to host the loaded kernel region (`loaded_kernel_pregion`) plus memory
453/// reserved for configuration.
454fn prepare_heap(
455    kernel_region: MemoryRegion<PhysAddr>,
456    loaded_kernel_pregion: MemoryRegion<PhysAddr>,
457    loaded_kernel_vregion: MemoryRegion<VirtAddr>,
458    platform: &dyn SvsmPlatform,
459    config: &SvsmConfig<'_>,
460) -> Result<KernelHeap, SvsmError> {
461    // Heap starts after kernel
462    let heap_pstart = loaded_kernel_pregion.end();
463    let heap_vstart = loaded_kernel_vregion.end();
464
465    // Compute size, excluding any memory reserved by the configuration.
466    let heap_size = kernel_region
467        .end()
468        .checked_sub(heap_pstart.into())
469        .and_then(|r| r.checked_sub(config.reserved_kernel_area_size()))
470        .expect("Insufficient physical space for kernel image")
471        .into();
472    let heap_pregion = MemoryRegion::new(heap_pstart, heap_size);
473    let heap_vregion = MemoryRegion::new(heap_vstart, heap_size);
474
475    // SAFETY: the virtual address range was computed so it is within the valid
476    // region and does not collide with the kernel.
477    unsafe {
478        map_and_validate(platform, config, heap_vregion, heap_pregion.start())?;
479    }
480
481    Ok(KernelHeap::create(heap_vstart, heap_pregion))
482}

484#[no_mangle]
485pub extern "C" fn stage2_main(launch_info: &Stage2LaunchInfo) -> ! {
486    let platform_type = SvsmPlatformType::from(launch_info.platform_type);
487
488    init_platform_type(platform_type);
489    let mut platform_cell = SvsmPlatformCell::new(true);
490    let platform = platform_cell.platform_mut();
491    let stage2_platform_cell = Stage2PlatformCell::new(platform_type);
492    let stage2_platform = stage2_platform_cell.platform();
493
494    // SAFETY: the address in the launch info is known to be correct.
495    let igvm_params = unsafe { IgvmParams::new(VirtAddr::from(launch_info.igvm_params as u64)) }
496        .expect("Failed to get IGVM parameters");
497    let config = SvsmConfig::new(&igvm_params);
498
499    // Set up space for an early IDT.  This will remain in scope as long as
500    // stage2 is in memory.
501    let mut early_idt = [IdtEntry::no_handler(); EARLY_IDT_ENTRIES];
502    let mut idt = IDT::new(&mut early_idt);
503
504    // Get a reference to the CPUID page if this platform requires it.
505    let cpuid_page = stage2_platform.get_cpuid_page(launch_info);
506
507    // SAFETY: the IDT here will remain in scope until the full IDT is
508    // initialized later, and thus can safely be used as the early IDT.
509    unsafe {
510        setup_env(&config, platform, launch_info, cpuid_page, &mut idt);
511    }
512
513    // Get the available physical memory region for the kernel
514    let kernel_region = config
515        .find_kernel_region()
516        .expect("Failed to find memory region for SVSM kernel");
517
518    log::info!("SVSM memory region: {kernel_region:#018x}");
519
520    init_valid_bitmap_alloc(kernel_region).expect("Failed to allocate valid-bitmap");
521
522    // The physical memory region we've loaded so far
523    let mut loaded_kernel_pregion = MemoryRegion::new(kernel_region.start(), 0);
524
525    // Load first the kernel ELF and update the loaded physical region
526    let (kernel_entry, loaded_kernel_vregion) =
527        load_kernel_elf(launch_info, &mut loaded_kernel_pregion, platform, &config)
528            .expect("Failed to load kernel ELF");
529
530    // Create the page heap used in the kernel region.
531    let mut kernel_heap = prepare_heap(
532        kernel_region,
533        loaded_kernel_pregion,
534        loaded_kernel_vregion,
535        platform,
536        &config,
537    )
538    .expect("Could not create kernel heap");
539
540    // Load the IGVM params, if present. Update loaded region accordingly.
541    // SAFETY: The loaded kernel region was correctly calculated above and
542    // is sized appropriately to include a copy of the IGVM parameters.
543    let igvm_vaddr = load_igvm_params(&mut kernel_heap, &config, launch_info)
544        .expect("Failed to load IGVM params");
545
546    // Determine whether this platforms uses a secrets pgae.
547    let secrets_page = stage2_platform.get_secrets_page(launch_info);
548
549    // Determine whether use of interrupts on the SVSM should be suppressed.
550    // This is required when running SNP under KVM/QEMU.
551    let suppress_svsm_interrupts = match platform_type {
552        SvsmPlatformType::Snp => config.suppress_svsm_interrupts_on_snp(),
553        _ => false,
554    };
555
556    // Build the handover information describing the memory layout and hand
557    // control to the SVSM kernel.
558    let launch_info = KernelLaunchInfo {
559        kernel_region_phys_start: u64::from(kernel_region.start()),
560        kernel_region_phys_end: u64::from(kernel_region.end()),
561        heap_area_phys_start: u64::from(kernel_heap.phys_base()),
562        heap_area_virt_start: u64::from(kernel_heap.virt_base()),
563        heap_area_page_count: kernel_heap.page_count().try_into().unwrap(),
564        heap_area_allocated: kernel_heap.next_free().try_into().unwrap(),
565        kernel_region_virt_start: u64::from(loaded_kernel_vregion.start()),
566        kernel_elf_stage2_virt_start: u64::from(launch_info.kernel_elf_start),
567        kernel_elf_stage2_virt_end: u64::from(launch_info.kernel_elf_end),
568        kernel_fs_start: u64::from(launch_info.kernel_fs_start),
569        kernel_fs_end: u64::from(launch_info.kernel_fs_end),
570        stage2_start: 0x800000u64,
571        stage2_end: launch_info.stage2_end as u64,
572        cpuid_page: u64::from(cpuid_page.unwrap_or(VirtAddr::null())),
573        secrets_page: u64::from(secrets_page.unwrap_or(VirtAddr::null())),
574        stage2_igvm_params_phys_addr: u64::from(launch_info.igvm_params),
575        stage2_igvm_params_size: igvm_params.size() as u64,
576        igvm_params_virt_addr: u64::from(igvm_vaddr),
577        vtom: launch_info.vtom,
578        debug_serial_port: config.debug_serial_port(),
579        use_alternate_injection: config.use_alternate_injection(),
580        suppress_svsm_interrupts,
581        platform_type,
582    };
583
584    check_launch_info(&launch_info);
585
586    let mem_info = memory_info();
587    print_memory_info(&mem_info);
588
589    log::info!(
590        "  kernel_region_phys_start = {:#018x}",
591        kernel_region.start()
592    );
593    log::info!("  kernel_region_phys_end   = {:#018x}", kernel_region.end());
594    log::info!(
595        "  kernel_virtual_base      = {:#018x}",
596        loaded_kernel_vregion.start()
597    );
598
599    let valid_bitmap = valid_bitmap_addr();
600
601    log::info!("Starting SVSM kernel...");
602
603    // SAFETY: the addreses used to invoke the kernel have been calculated
604    // correctly for use in the assembly trampoline.
605    unsafe {
606        // Shut down the PerCpu instance
607        shutdown_percpu();
608
609        asm!("jmp *%rax",
610             in("rax") u64::from(kernel_entry),
611             in("rdi") &launch_info,
612             in("rsi") valid_bitmap.bits(),
613             options(att_syntax))
614    };
615
616    unreachable!("Road ends here!");
617}
