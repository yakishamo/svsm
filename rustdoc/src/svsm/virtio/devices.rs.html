8use super::hal::*;
9extern crate alloc;
10use alloc::boxed::Box;
11use core::ptr::NonNull;
12use virtio_drivers::device::blk::VirtIOBlk;
13use virtio_drivers::transport::mmio::{MmioError, MmioTransport};
14use virtio_drivers::transport::{DeviceType, Transport};
15use virtio_drivers::PAGE_SIZE;
16
17use super::error::*;
18use crate::address::PhysAddr;
19use crate::error::SvsmError;
20use crate::locking::SpinLock;
21use crate::mm::global_memory::{map_global_range_4k_shared, GlobalRangeGuard};
22use crate::mm::pagetable::PTEntryFlags;
23
24pub struct VirtIOBlkDevice {
25    pub device: SpinLock<VirtIOBlk<SvsmHal, MmioTransport<SvsmHal>>>,
26    _mmio_space: GlobalRangeGuard,
27}
28
29impl core::fmt::Debug for VirtIOBlkDevice {
30    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
31        f.debug_struct("VirtIOBlkDevice").finish()
32    }
33}
34
35impl VirtIOBlkDevice {
36    pub fn new(mmio_base: PhysAddr) -> Result<Box<Self>, SvsmError> {
37        virtio_init();
38
39        let mem = map_global_range_4k_shared(mmio_base, PAGE_SIZE, PTEntryFlags::data())?;
40
41        // Not expected to fail, because mem exists.
42        let header = NonNull::new(mem.addr().as_mut_ptr()).unwrap();
43
44        // SAFETY: `header` is the MMIO config area; we have to trust the content is valid.
45        let transport = unsafe {
46            // TODO: Use more detailed error types ?
47            MmioTransport::<SvsmHal>::new(header).map_err(|e| match e {
48                MmioError::BadMagic(_) => VirtioError::InvalidDevice,
49                MmioError::UnsupportedVersion(_) => VirtioError::InvalidDevice,
50                MmioError::ZeroDeviceId => VirtioError::InvalidDevice,
51            })?
52        };
53
54        if transport.device_type() != DeviceType::Block {
55            return Err(VirtioError::InvalidDeviceType)?;
56        }
57
58        let blk = VirtIOBlk::new(transport).map_err(|_| VirtioError::InvalidDevice)?;
59
60        Ok(Box::new(VirtIOBlkDevice {
61            device: SpinLock::new(blk),
62            _mmio_space: mem,
63        }))
64    }
65}
