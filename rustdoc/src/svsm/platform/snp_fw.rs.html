44fn validate_fw_mem_region(
45    config: &SvsmConfig<'_>,
46    region: MemoryRegion<PhysAddr>,
47) -> Result<(), SvsmError> {
48    let pstart = region.start();
49    let pend = region.end();
50
51    log::info!("Validating {:#018x}-{:#018x}", pstart, pend);
52
53    if config.page_state_change_required() {
54        current_ghcb()
55            .page_state_change(region, PageSize::Regular, PageStateChangeOp::Private)
56            .expect("GHCB PSC call failed to validate firmware memory");
57    }
58
59    for paddr in region.iter_pages(PageSize::Regular) {
60        let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
61        let vaddr = guard.virt_addr();
62
63        // SAFETY: the virtual address mapping is known to point to the guest
64        // physical address range supplied by the caller.
65        unsafe {
66            pvalidate(vaddr, PageSize::Regular, PvalidateOp::Valid)?;
67
68            // Make page accessible to guest VMPL
69            rmp_adjust(
70                vaddr,
71                RMPFlags::GUEST_VMPL | RMPFlags::RWX,
72                PageSize::Regular,
73            )?;
74
75            zero_mem_region(vaddr, vaddr + PAGE_SIZE);
76        }
77    }
78
79    Ok(())
80}

82fn validate_fw_memory_vec(
83    config: &SvsmConfig<'_>,
84    regions: Vec<MemoryRegion<PhysAddr>>,
85) -> Result<(), SvsmError> {
86    if regions.is_empty() {
87        return Ok(());
88    }
89
90    let mut next_vec = Vec::new();
91    let mut region = regions[0];
92
93    for next in regions.into_iter().skip(1) {
94        if region.contiguous(&next) {
95            region = region.merge(&next);
96        } else {
97            next_vec.push(next);
98        }
99    }
100
101    validate_fw_mem_region(config, region)?;
102    validate_fw_memory_vec(config, next_vec)
103}

105pub fn validate_fw_memory(
106    config: &SvsmConfig<'_>,
107    fw_meta: &SevFWMetaData,
108    kernel_region: &MemoryRegion<PhysAddr>,
109) -> Result<(), SvsmError> {
110    // Initalize vector with regions from the FW
111    let mut regions = fw_meta.valid_mem.clone();
112
113    // Add region for CPUID page if present
114    if let Some(cpuid_paddr) = fw_meta.cpuid_page {
115        regions.push(MemoryRegion::new(cpuid_paddr, PAGE_SIZE));
116    }
117
118    // Add region for Secrets page if present
119    if let Some(secrets_paddr) = fw_meta.secrets_page {
120        regions.push(MemoryRegion::new(secrets_paddr, PAGE_SIZE));
121    }
122
123    // Add region for CAA page if present
124    if let Some(caa_paddr) = fw_meta.caa_page {
125        regions.push(MemoryRegion::new(caa_paddr, PAGE_SIZE));
126    }
127
128    // Sort regions by base address
129    regions.sort_unstable_by_key(|a| a.start());
130
131    for region in regions.iter() {
132        if region.overlap(kernel_region) {
133            log::error!("FwMeta region ovelaps with kernel");
134            return Err(SvsmError::Firmware);
135        }
136    }
137
138    validate_fw_memory_vec(config, regions)
139}

143
144    match fw_meta.cpuid_page {
145        Some(addr) => log::info!("  CPUID Page   : {:#010x}", addr),
146        None => log::info!("  CPUID Page   : None"),

155        Some(addr) => log::info!("  CAA Page     : {:#010x}", addr),
156        None => log::info!("  CAA Page     : None"),
157    };
158
159    for region in &fw_meta.valid_mem {
160        log::info!("  Pre-Validated Region {region:#018x}");
161    }
162}

164fn copy_cpuid_table_to_fw(fw_addr: PhysAddr) -> Result<(), SvsmError> {
165    let guard = PerCPUPageMappingGuard::create_4k(fw_addr)?;
166
167    // SAFETY: this is called from CPU 0, so the underlying physical address
168    // is not being aliased. We are mapping a full page, which is 4K-aligned,
169    // and is enough for SnpCpuidTable.
170    unsafe {
171        copy_cpuid_table_to(guard.virt_addr());
172    }
173
174    Ok(())
175}

177fn copy_secrets_page_to_fw(
178    fw_addr: PhysAddr,
179    caa_addr: PhysAddr,
180    kernel_region: &MemoryRegion<PhysAddr>,
181) -> Result<(), SvsmError> {
182    let guard = PerCPUPageMappingGuard::create_4k(fw_addr)?;
183    let start = guard.virt_addr();
184
185    // Zero target
186    // SAFETY: we trust PerCPUPageMappingGuard::create_4k() to return a
187    // valid pointer to a correctly mapped region of size PAGE_SIZE.
188    unsafe {
189        zero_mem_region(start, start + PAGE_SIZE);
190    }
191
192    // Copy secrets page
193    let mut fw_secrets_page = secrets_page().unwrap().copy_for_vmpl(GUEST_VMPL);
194
195    fw_secrets_page.set_svsm_data(
196        kernel_region.start().into(),
197        kernel_region.len().try_into().unwrap(),
198        u64::from(caa_addr),
199    );
200
201    // SAFETY: start points to a new allocated and zeroed page.
202    unsafe {
203        fw_secrets_page.copy_to(start);
204    }
205
206    Ok(())
207}

209fn zero_caa_page(fw_addr: PhysAddr) -> Result<(), SvsmError> {
210    let guard = PerCPUPageMappingGuard::create_4k(fw_addr)?;
211    let vaddr = guard.virt_addr();
212
213    // SAFETY: we trust PerCPUPageMappingGuard::create_4k() to return a
214    // valid pointer to a correctly mapped region of size PAGE_SIZE.
215    unsafe {
216        zero_mem_region(vaddr, vaddr + PAGE_SIZE);
217    }
218
219    Ok(())
220}

222pub fn copy_tables_to_fw(
223    fw_meta: &SevFWMetaData,
224    kernel_region: &MemoryRegion<PhysAddr>,
225) -> Result<(), SvsmError> {
226    if let Some(addr) = fw_meta.cpuid_page {
227        copy_cpuid_table_to_fw(addr)?;
228    }
229
230    let secrets_page = fw_meta.secrets_page.ok_or(SvsmError::MissingSecrets)?;
231    let caa_page = fw_meta.caa_page.ok_or(SvsmError::MissingCAA)?;
232
233    copy_secrets_page_to_fw(secrets_page, caa_page, kernel_region)?;
234
235    zero_caa_page(caa_page)?;
236
237    Ok(())
238}

240pub fn validate_fw(
241    config: &SvsmConfig<'_>,
242    kernel_region: &MemoryRegion<PhysAddr>,
243) -> Result<(), SvsmError> {
244    let flash_regions = config.get_fw_regions(kernel_region);
245
246    for (i, region) in flash_regions.into_iter().enumerate() {
247        log::info!(
248            "Flash region {} at {:#018x} size {:018x}",
249            i,
250            region.start(),
251            region.len(),
252        );
253
254        for paddr in region.iter_pages(PageSize::Regular) {
255            let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
256            let vaddr = guard.virt_addr();
257            // SAFETY: the address is known to be a guest page.
258            if let Err(e) = unsafe {
259                rmp_adjust(
260                    vaddr,
261                    RMPFlags::GUEST_VMPL | RMPFlags::RWX,
262                    PageSize::Regular,
263                )
264            } {
265                log::info!("rmpadjust failed for addr {:#018x}", vaddr);
266                return Err(e);
267            }
268        }
269    }
270
271    Ok(())
272}

274pub fn prepare_fw_launch(fw_meta: &SevFWMetaData) -> Result<(), SvsmError> {
275    if let Some(caa) = fw_meta.caa_page {
276        this_cpu_shared().update_guest_caa(caa);
277    }
278
279    this_cpu().alloc_guest_vmsa()?;
280    this_cpu().update_guest_mappings()?;
281
282    Ok(())
283}

285pub fn launch_fw(config: &SvsmConfig<'_>) -> Result<(), SvsmError> {
286    let cpu = this_cpu();
287    let mut vmsa_ref = cpu.guest_vmsa_ref();
288    let vmsa_pa = vmsa_ref.vmsa_phys().unwrap();
289    let vmsa = vmsa_ref.vmsa();
290
291    config.initialize_guest_vmsa(vmsa)?;
292
293    log::info!("VMSA PA: {:#x}", vmsa_pa);
294
295    let sev_features = vmsa.sev_features;
296
297    log::info!("Launching Firmware");
298    current_ghcb().register_guest_vmsa(vmsa_pa, 0, GUEST_VMPL as u64, sev_features)?;
299
300    Ok(())
301}
