20#[derive(Clone, Copy, Debug, Default)]
21struct StackFrame {
22    rbp: VirtAddr,
23    rsp: VirtAddr,
24    rip: VirtAddr,
25    is_aligned: bool,
26    is_last: bool,
27    is_exception_frame: bool,
28    _stack_depth: usize, // Not needed for frame unwinding, only as diagnostic information.
29}
30
31#[derive(Clone, Copy, Debug)]
32enum UnwoundStackFrame {
33    Valid(StackFrame),
34    Invalid,
35}
36
37type StacksBounds = [MemoryRegion<VirtAddr>; 3];
38
39#[derive(Debug)]
40struct StackUnwinder {
41    next_frame: Option<UnwoundStackFrame>,
42    stacks: StacksBounds,
43}
44
45extern "C" {
46    static bsp_stack: u8;
47    static bsp_stack_end: u8;
48}
49
50fn is_stage2() -> bool {
51    // If the default BSP stack lands under 16MB, we're in Stage2.
52    (&raw const bsp_stack_end as usize) < (16 << 20)
53}
54
55impl StackUnwinder {
56    pub fn unwind_this_cpu() -> Self {
57        let mut rbp: usize;
58        // SAFETY: Inline assembly to read RBP, which does not change any state
59        // related to memory safety.
60        unsafe {
61            asm!("movq %rbp, {}", out(reg) rbp,
62                 options(att_syntax));
63        };
64
65        let stacks: StacksBounds = if let Some(cpu) = try_this_cpu() {
66            let current_stack = cpu.get_current_stack();
67            let cs_stack = cpu
68                .get_top_of_context_switch_stack()
69                .map_or(MemoryRegion::new(VirtAddr::null(), 0), |tos| {
70                    MemoryRegion::from_addresses(tos - STACK_SIZE, tos)
71                });
72            let df_stack = cpu
73                .get_top_of_df_stack()
74                .map_or(MemoryRegion::new(VirtAddr::null(), 0), |tos| {
75                    MemoryRegion::from_addresses(tos - STACK_SIZE, tos)
76                });
77            [current_stack, cs_stack, df_stack]
78        } else {
79            // Use default stack addresses.
80            if is_stage2() {
81                let bsp_init_stack = MemoryRegion::from_addresses(
82                    VirtAddr::from(STAGE2_STACK_END as u64),
83                    VirtAddr::from(STAGE2_STACK as u64),
84                );
85                let no_stack = MemoryRegion::new(VirtAddr::null(), 0);
86                [bsp_init_stack, no_stack, no_stack]
87            } else {
88                let bsp_init_stack = MemoryRegion::from_addresses(
89                    VirtAddr::from(&raw const bsp_stack),
90                    VirtAddr::from(&raw const bsp_stack_end),
91                );
92                let cs_stack = MemoryRegion::new(SVSM_CONTEXT_SWITCH_STACK, STACK_TOTAL_SIZE);
93                let df_stack = MemoryRegion::new(SVSM_STACK_IST_DF_BASE, STACK_TOTAL_SIZE);
94                [bsp_init_stack, cs_stack, df_stack]
95            }
96        };
97
98        Self::new(VirtAddr::from(rbp), stacks)
99    }
100
101    fn new(rbp: VirtAddr, stacks: StacksBounds) -> Self {
102        let first_frame = Self::unwind_framepointer_frame(rbp, &stacks);
103        Self {
104            next_frame: Some(first_frame),
105            stacks,
106        }
107    }
108
109    fn check_unwound_frame(
110        rbp: VirtAddr,
111        rsp: VirtAddr,
112        rip: VirtAddr,
113        stacks: &StacksBounds,
114    ) -> UnwoundStackFrame {
115        // The next frame's rsp or rbp should live on some valid stack,
116        // otherwise mark the unwound frame as invalid.
117        let Some(stack) = stacks.iter().find(|stack| {
118            !stack.is_empty() && (stack.contains_inclusive(rsp) || stack.contains_inclusive(rbp))
119        }) else {
120            log::info!("check_unwound_frame: rsp {rsp:#018x} and rbp {rbp:#018x} does not match any known stack");
121            return UnwoundStackFrame::Invalid;
122        };
123
124        // The x86-64 ABI requires stack frames to be 16b-aligned
125        let is_aligned = rbp.is_aligned(16);
126        let is_last = Self::frame_is_last(rbp);
127        let is_exception_frame = is_exception_handler_return_site(rip);
128
129        if !is_last && !is_exception_frame {
130            // Consistency check to ensure forward-progress: never unwind downwards.
131            if rbp < rsp {
132                return UnwoundStackFrame::Invalid;
133            }
134        }
135
136        let _stack_depth = stack.end() - rsp;
137
138        UnwoundStackFrame::Valid(StackFrame {
139            rbp,
140            rsp,
141            rip,
142            is_aligned,
143            is_last,
144            is_exception_frame,
145            _stack_depth,
146        })
147    }
148
149    fn unwind_framepointer_frame(rbp: VirtAddr, stacks: &StacksBounds) -> UnwoundStackFrame {
150        let rsp = rbp;
151
152        // Storage for return address + saved %rbp
153        let Some(range) = MemoryRegion::checked_new(rsp, 2 * mem::size_of::<VirtAddr>()) else {
154            return UnwoundStackFrame::Invalid;
155        };
156
157        if !stacks.iter().any(|stack| stack.contains_region(&range)) {
158            return UnwoundStackFrame::Invalid;
159        }
160
161        // Saved %rbp
162        //
163        // SAFETY: This function always works on the stacks of the current
164        // context, so de-referencing pointers from the stacks of the context
165        // is safe.
166        let rbp = unsafe { rsp.as_ptr::<VirtAddr>().read_unaligned() };
167        let rsp = rsp + mem::size_of::<VirtAddr>();
168        // Return address
169        //
170        // SAFETY: This function always works on the stacks of the current
171        // context, so de-referencing pointers from the stacks of the context
172        // is safe.
173        let rip = unsafe { rsp.as_ptr::<VirtAddr>().read_unaligned() };
174        let rsp = rsp + mem::size_of::<VirtAddr>();
175
176        Self::check_unwound_frame(rbp, rsp, rip, stacks)
177    }
178
179    fn unwind_exception_frame(rsp: VirtAddr, stacks: &StacksBounds) -> UnwoundStackFrame {
180        let Some(range) = MemoryRegion::checked_new(rsp, mem::size_of::<X86ExceptionContext>())
181        else {
182            return UnwoundStackFrame::Invalid;
183        };
184
185        if !stacks.iter().any(|stack| stack.contains_region(&range)) {
186            return UnwoundStackFrame::Invalid;
187        }
188
189        // SAFETY: rsp is in a valid memory range as checked previously
190        // in this function. It is always properly aligned because
191        // X86ExceptionContext is packed(1). It's in the per-CPU stack
192        // so no aliasing can occur.
193        let Some(ctx) = (unsafe { rsp.as_ptr::<X86ExceptionContext>().as_ref() }) else {
194            return UnwoundStackFrame::Invalid;
195        };
196        let rbp = VirtAddr::from(ctx.regs.rbp);
197        let rip = VirtAddr::from(ctx.frame.rip);
198        let rsp = VirtAddr::from(ctx.frame.rsp);
199
200        Self::check_unwound_frame(rbp, rsp, rip, stacks)
201    }
202
203    fn frame_is_last(rbp: VirtAddr) -> bool {
204        // A new task is launched with RBP = 0, which is pushed onto the stack
205        // immediately and can serve as a marker when the end of the stack has
206        // been reached.
207        rbp == VirtAddr::null()
208    }
209}
210
211impl Iterator for StackUnwinder {
212    type Item = UnwoundStackFrame;
213
214    fn next(&mut self) -> Option<Self::Item> {
215        let cur = self.next_frame;
216        match cur {
217            Some(cur) => {
218                match &cur {
219                    UnwoundStackFrame::Invalid => {
220                        self.next_frame = None;
221                    }
222                    UnwoundStackFrame::Valid(cur_frame) => {
223                        if cur_frame.is_last {
224                            self.next_frame = None
225                        } else if cur_frame.is_exception_frame {
226                            self.next_frame =
227                                Some(Self::unwind_exception_frame(cur_frame.rsp, &self.stacks));
228                        } else {
229                            self.next_frame =
230                                Some(Self::unwind_framepointer_frame(cur_frame.rbp, &self.stacks));
231                        }
232                    }
233                };
234
235                Some(cur)
236            }
237            None => None,
238        }
239    }
240}

242fn print_stack_frame(frame: StackFrame) {
243    let mut annotated = false;
244    let mut msg = format!("  [{:016x}]", frame.rip);
245
246    if frame.is_exception_frame {
247        msg.push_str(" @");
248        annotated = true;
249    }
250    if !frame.is_aligned {
251        msg.push_str(if annotated { "#" } else { " #" });
252    }
253    log::info!("{}", msg);
254}
255
256pub fn print_stack(skip: usize) {
257    let unwinder = StackUnwinder::unwind_this_cpu();
258    log::info!("---BACKTRACE---:");
259    for frame in unwinder.skip(skip) {
260        match frame {
261            UnwoundStackFrame::Valid(item) => print_stack_frame(item),
262            UnwoundStackFrame::Invalid => log::info!("  Invalid frame"),
263        }
264    }
265    log::info!("---END---");
266}
