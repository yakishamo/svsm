18pub fn init_page_table(
19    launch_info: &KernelLaunchInfo,
20    kernel_elf: &elf::Elf64File<'_>,
21) -> Result<PageBox<PageTable>, SvsmError> {
22    let mut pgtable = PageTable::allocate_new()?;
23
24    // Install mappings for the kernel's ELF segments each.
25    // The memory backing the kernel ELF segments gets allocated back to back
26    // from the physical memory region by the Stage2 loader.
27    let mut phys = PhysAddr::from(launch_info.kernel_region_phys_start);
28    for segment in kernel_elf.image_load_segment_iter(launch_info.kernel_region_virt_start) {
29        let vaddr_start = VirtAddr::from(segment.vaddr_range.vaddr_begin);
30        let vaddr_end = VirtAddr::from(segment.vaddr_range.vaddr_end);
31        let aligned_vaddr_end = vaddr_end.page_align_up();
32        let segment_len = aligned_vaddr_end - vaddr_start;
33        let flags = if segment.flags.contains(elf::Elf64PhdrFlags::EXECUTE) {
34            PTEntryFlags::exec()
35        } else if segment.flags.contains(elf::Elf64PhdrFlags::WRITE) {
36            PTEntryFlags::data()
37        } else {
38            PTEntryFlags::data_ro()
39        };
40
41        let vregion = MemoryRegion::new(vaddr_start, segment_len);
42        pgtable
43            .map_region(vregion, phys, flags)
44            .expect("Failed to map kernel ELF segment");
45
46        phys = phys + segment_len;
47    }
48
49    // Map subsequent heap area.
50    let heap_vregion = MemoryRegion::new(
51        VirtAddr::from(launch_info.heap_area_virt_start),
52        launch_info.heap_area_page_count as usize * PAGE_SIZE,
53    );
54    pgtable
55        .map_region(
56            heap_vregion,
57            PhysAddr::from(launch_info.heap_area_phys_start),
58            PTEntryFlags::data(),
59        )
60        .expect("Failed to map heap");
61
62    init_global_ranges();
63
64    Ok(pgtable)
65}

67fn invalidate_boot_memory_region(
68    platform: &dyn SvsmPlatform,
69    config: &SvsmConfig<'_>,
70    region: MemoryRegion<PhysAddr>,
71) -> Result<(), SvsmError> {
72    // Caller must ensure the memory region's starting address is page-aligned
73    let aligned_region = MemoryRegion::new(region.start(), page_align_up(region.len()));
74    log::info!("Invalidating boot region {aligned_region:#018x}");
75
76    if !aligned_region.is_empty() {
77        platform.validate_physical_page_range(aligned_region, PageValidateOp::Invalidate)?;
78
79        if config.page_state_change_required() {
80            platform.page_state_change(
81                aligned_region,
82                PageSize::Regular,
83                PageStateChangeOp::Shared,
84            )?;
85        }
86    }
87
88    Ok(())
89}

91pub fn invalidate_early_boot_memory(
92    platform: &dyn SvsmPlatform,
93    config: &SvsmConfig<'_>,
94    launch_info: &KernelLaunchInfo,
95) -> Result<(), SvsmError> {
96    // Early boot memory must be invalidated after changing to the SVSM page
97    // page table to avoid invalidating page tables currently in use.  Always
98    // invalidate stage 2 memory, unless firmware is loaded into low memory.
99    // Also invalidate the boot data if required.
100    if !config.fw_in_low_memory() {
101        let lowmem_region = MemoryRegion::from_addresses(
102            PhysAddr::from(0u64),
103            PhysAddr::from(u64::from(LOWMEM_END)),
104        );
105        invalidate_boot_memory_region(platform, config, lowmem_region)?;
106    }
107
108    let stage2_base = PhysAddr::from(launch_info.stage2_start);
109    let stage2_end = PhysAddr::from(launch_info.stage2_end);
110    let stage2_region = MemoryRegion::from_addresses(stage2_base, stage2_end);
111    invalidate_boot_memory_region(platform, config, stage2_region)?;
112
113    let kernel_elf_size =
114        launch_info.kernel_elf_stage2_virt_end - launch_info.kernel_elf_stage2_virt_start;
115    let kernel_elf_region = MemoryRegion::new(
116        PhysAddr::new(launch_info.kernel_elf_stage2_virt_start.try_into().unwrap()),
117        kernel_elf_size.try_into().unwrap(),
118    );
119    invalidate_boot_memory_region(platform, config, kernel_elf_region)?;
120
121    let kernel_fs_size = launch_info.kernel_fs_end - launch_info.kernel_fs_start;
122    if kernel_fs_size > 0 {
123        let kernel_fs_region = MemoryRegion::new(
124            PhysAddr::new(launch_info.kernel_fs_start.try_into().unwrap()),
125            kernel_fs_size.try_into().unwrap(),
126        );
127        invalidate_boot_memory_region(platform, config, kernel_fs_region)?;
128    }
129
130    if launch_info.stage2_igvm_params_size > 0 {
131        let igvm_params_region = MemoryRegion::new(
132            PhysAddr::new(launch_info.stage2_igvm_params_phys_addr.try_into().unwrap()),
133            launch_info.stage2_igvm_params_size as usize,
134        );
135        invalidate_boot_memory_region(platform, config, igvm_params_region)?;
136    }
137
138    Ok(())
139}
